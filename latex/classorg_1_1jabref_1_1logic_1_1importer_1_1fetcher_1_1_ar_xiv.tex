\begin{DoxyCode}{0}

\begin{DoxyCode}{0}
\DoxyCodeLine{searchQuery }

\begin{DoxyCode}{0}
\DoxyCodeLine{ids }
\end{DoxyCode}

\begin{DoxyCode}{0}

\end{DoxyCode}
 . This allows the A\+PI to act as a results filter.
